## Application Entry Point

```tsx title="resources/js/app.tsx"
import { createInertiaApp } from '@inertiajs/react';
import { resolvePageComponent } from 'laravel-vite-plugin/inertia-helpers';
import { StrictMode } from 'react';
import { createRoot } from 'react-dom/client';
import { initializeTheme } from './hooks/use-appearance';

const appName = import.meta.env.VITE_APP_NAME || 'Laravel';

createInertiaApp({
    title: (title) => (title ? `${title} - ${appName}` : appName),
    
    // Dynamically resolve page components
    resolve: (name) =>
        resolvePageComponent(
            `./pages/${name}.tsx`,
            import.meta.glob('./pages/**/*.tsx'),
        ),
    
    setup({ el, App, props }) {
        const root = createRoot(el);
        root.render(
            <StrictMode>
                <App {...props} />
            </StrictMode>
        );
    },
    
    progress: {
        color: '#4B5563',  // Loading bar color
    },
});

// Initialize dark/light mode
initializeTheme();
```

**Key Features:**
- Dynamic page component resolution with code splitting
- React 19 Strict Mode for development warnings
- Progress bar for navigation transitions
- Theme initialization on load

## Inertia.js Integration

### How Inertia Works

Inertia.js bridges Laravel and React without building a separate API:

```
1. User clicks link/submits form
   ↓
2. Inertia intercepts and makes XHR request
   ↓
3. Laravel controller processes request
   ↓
4. Controller returns Inertia::render('Page', $props)
   ↓
5. Inertia sends JSON response with component + props
   ↓
6. React renders new component with props
   ↓
7. Browser URL updates (no full reload)
```

### Shared Props

Data available to all pages via middleware:

```tsx
import { usePage } from '@inertiajs/react';

function Component() {
```

### Zereginak (Tasks)

Task management with calendar view using `react-big-calendar`.

```tsx title="pages/Zereginak/index.tsx" (conceptual)
import { Calendar, momentLocalizer } from 'react-big-calendar';
import moment from 'moment';
import 'react-big-calendar/lib/css/react-big-calendar.css';

const localizer = momentLocalizer(moment);

interface Props {
    zereginak: Zereginak[];
}

export default function ZereginakIndex({ zereginak }: Props) {
    // Transform tasks to calendar events
    const events = zereginak.flatMap(task =>
        task.erabiltzaileak.map(user => ({
            title: task.izena,
            start: new Date(user.pivot.hasiera_data),
            end: new Date(user.pivot.amaiera_data),
            resource: { task, user },
        }))
    );

    const handleSelectEvent = (event) => {
        router.visit(`/zereginak/${event.resource.task.id}/editatu`);
    };

    return (
        <AppLayout>
            <Head title="Tasks" />
            
            <div className="h-screen">
                <Calendar
                    localizer={localizer}
                    events={events}
                    onSelectEvent={handleSelectEvent}
                    views={['month', 'week', 'day']}
                    defaultView="month"
                />
            </div>
            
            <Link href="/zereginak/sortu" className="btn btn-primary">
                Create Task
            </Link>
        </AppLayout>
    );
}
```

## Custom Hooks

### useAppearance (Dark Mode)

```tsx title="hooks/use-appearance.ts"
import { useEffect, useState } from 'react';

type Appearance = 'light' | 'dark' | 'system';

export function useAppearance() {
    const [appearance, setAppearance] = useState<Appearance>(
        () => (localStorage.getItem('appearance') as Appearance) || 'system'
    );

    useEffect(() => {
        const root = document.documentElement;
        const systemPrefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
        
        const isDark = appearance === 'dark' || 
                      (appearance === 'system' && systemPrefersDark);
        
        root.classList.toggle('dark', isDark);
        localStorage.setItem('appearance', appearance);
    }, [appearance]);

    return { appearance, setAppearance };
}

export function initializeTheme() {
    const appearance = localStorage.getItem('appearance') || 'system';
    const systemPrefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
    const isDark = appearance === 'dark' || (appearance === 'system' && systemPrefersDark);
    
    document.documentElement.classList.toggle('dark', isDark);
}
```

## State Management

7POINTS.NET uses **server-driven state** via Inertia.js instead of client-side state management libraries (Redux, Zustand, etc.).

### Why Server-Driven State?

- **Single source of truth** - Database is the state
- **No synchronization issues** - Server always sends current data
- **Simpler architecture** - No need for actions, reducers, store
- **Better for SEO** - Server-rendered initial state

### Client State (React Hooks)

Only UI-specific state is managed client-side:

```tsx
// Form state
const { data, setData } = useForm({ ... });

// UI state
const [isModalOpen, setIsModalOpen] = useState(false);
const [selectedMonth, setSelectedMonth] = useState('');

// Derived state
const totalExpenses = useMemo(
    () => gastos.reduce((sum, g) => sum + parseFloat(g.Cantidad), 0),
    [gastos]
);
```
